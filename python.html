def fibonacci(n):
	if n == 0:
	    return 0
	elif n == 1:
	    return 1
	else:
	    return fibonacci(n - 1) + fibonacci(n - 2)

import os

def copy_file():
		source_file = input("Enter the name of the source file: ")# Check if the source file exists
		if not os.path.exists(source_file):
		    print("Error: File does not exist.")
		    return
		
		# Prompt user for the destination file name
		dest_file = input("Enter the name of the destination file: ")
		
		# Open source file and read contents
		with open(source_file, 'r') as f:
		    file_contents = f.read()
		
		# Write contents to destination file
		with open(dest_file, 'w') as f:
		    f.write(file_contents)
		
		print(f"Contents of {source_file} have been copied to {dest_file}.")

Write a program by using Labels as Output Fields

import tkinter as tk

# Create the main window
root = tk.Tk()
root.title("Label Output Example")

# Create input fields
tk.Label(root, text="Enter your name:").grid(row=0, column=0)
name_entry = tk.Entry(root)
name_entry.grid(row=0, column=1)

tk.Label(root, text="Enter your age:").grid(row=1, column=0)
age_entry = tk.Entry(root)
age_entry.grid(row=1, column=1)

# Create output labels
tk.Label(root, text="Name:").grid(row=2, column=0)
name_label = tk.Label(root, text="")
name_label.grid(row=2, column=1)

tk.Label(root, text="Age:").grid(row=3, column=0)
age_label = tk.Label(root, text="")
age_label.grid(row=3, column=1)

# Create button to update labels
def update_labels():
    name = name_entry.get()
    age = age_entry.get()
    name_label.config(text=name)
    age_label.config(text=age)

tk.Button(root, text="Update", command=update_labels).grid(row=4, column=1)

root.mainloop()

Reverse a string word by word

class ReverseString:
		def reverse_words(self, s: str) -> str:

	    words = s.split()
	
	    reversed_string = ' '.join(words[::-1])

    return reversed_string
